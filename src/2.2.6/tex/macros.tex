% svg inclusion
\newcommand{\svg}[3][0.7] {
  \begin{figure}[h]
    \begin{center}
      \fontsize{7}{8}\selectfont
      \includesvg[scale=#1]{\picdir #2}
      \caption{#3}
    \end{center}
  \end{figure}
}

% Simple single-line alignment
\newcommand{\salign}[2][] {
  \begin{align#1}
    #2
  \end{align#1}
}

% Таблицы
\newcommand{\stbl}[2] {
  \begin{table}[h]
    \begin{center}
      \input{\tbldir #1}
      \caption{#2}
    \end{center}
  \end{table}
}
